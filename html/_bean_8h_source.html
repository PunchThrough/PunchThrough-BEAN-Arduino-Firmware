9 /*
10  * An accelerometer interrupt type, they are as follows:
11  * FLAT_EVENT - triggers when the accelerometer is lying flat on a surface
12  * ORIENT_EVENT - triggers when the accelerometer is NOT lying flat on a surface, but is tilted * in any direction
13  * SINGLE_TAP_EVENT - triggers when the accelerometer is tapped once
14  * DOUBLE_TAP_EVENT - triggers when the accelerometer is tapped twice
15  * ANY_MOTION_EVENT - triggers when the accelerometer experiences any change in motion
16  * HIGH_G_EVENT - triggers when the accelerometer experiences a velocity event higher than it's * sensitivity
17  * LOW_G_EVENT - triggers when the accelerometer is in free fall or experiences no gravitational * pull
18  */
19 typedef enum AccelEventTypes {
20  FLAT_EVENT = 0x80,
21  ORIENT_EVENT = 0x40,
22  SINGLE_TAP_EVENT = 0x20,
23  DOUBLE_TAP_EVENT = 0x10,
24  ANY_MOTION_EVENT = 0x04,
25  HIGH_G_EVENT = 0x02,
26  LOW_G_EVENT = 0x01
27 };
